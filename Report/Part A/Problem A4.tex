\bigskip
\subsection{Problem A4}\label{sec:qa4}\hfill

\medskip
\noindent Defining deviation variables: $\bar{x}_1 = x_{1} - x_{1}^e$, $\bar{x}_2 = x_{2} - x_{2}^e$, $\bar{I} = I - I^e$, $\bar{V} = V - V^e$, $\underbar{$\bar{x}$} = \underbar{$x$} - \underbar{$x$}^e$, $\underbar{$\dot{\bar{x}}$} = \underbar{$\dot{x}$}$.\\
\noindent Subtract equation \eqref{a3:equilibrium_1} from equation \eqref{a2:ssr}:
\begin{align} \label{a4:pre-linear}
    \underbar{$\dot{x}$}
    =
    \begin{bmatrix}
        x_2 - x_{2}^e \cr
        \left( \frac{5}{7m} \right)
        \left( c \left( \frac{I^2}{ \left( \delta - x_1 \right)^2} - \frac{ \left(I^{e} \right)^2}{ \left( \delta - x_{1}^e \right)^2 } \right)
        - k \left(x_{1} - x_{1}^e \right)
        - b \left(x_{2} - x_{2}^e \right) \right)\cr
        \frac{V-IR}{L_0 + L_1 e^{- \alpha \left( \delta - x_1 \right)}} \cr
    \end{bmatrix}
\end{align}
\noindent Use Taylor's Theorem to linearise $\underbar{$\dot{x}$}$ at $\left( \underbar{$x$}^e, V^{e} \right)$. Let $\psi = \frac{I^2}{ \left( \delta - x_1 \right)^2}$ and $\lambda =\frac{\zeta \beta}{L_0 + L_1 e^{- \alpha \left( \delta - x_1 \right)}}$, where $\zeta$ is a constant (1 or $R$) and $\beta$ is a variable ($V$ or $I$).\\
\noindent Linearise $\psi$:
\begin{equation} \label{a4:taylor_eqn_phi}
    \psi \left( I, x_{1} \right) \approx 
    \psi \left( I^{e}, x_{1}^{e} \right)
    + \frac{\partial \psi}{\partial I}
    \Bigr|_{\substack{I^{e}, x_{1}^{e}}}
     \left( I - I^e \right)  
    + \frac{\partial \psi}{\partial x_1}
    \Bigr|_{\substack{I^{e}, x_{1}^{e}}} (x_{1} - x_{1}^{e} )
\end{equation}
 
\medskip
\noindent Find the partial derivatives:
\begin{align} \label{a4:partial_psi_i}
    \frac{\partial \psi}{\partial I} \Bigr|_{\substack{I^{e}, x_{1}^{e}}}
    &= \frac{\partial}{\partial I} \left ( \frac{I^2}{ \left( \delta - x_1 \right)^2} \right ) \Bigr|_{\substack{I^{e}, x_{1}^{e}}}
    = \frac{2 I}{ \left( \delta - x_1 \right)^2} \Bigr|_{\substack{I^{e}, x_{1}^{e}}}
    = \frac{2 I^e}{ \left( \delta - x_1^e \right)^2}
\end{align}
\begin{align} \label{a4:partial_psi_x1}
    \frac{\partial \psi}{\partial x_1} 
    \Bigr|_{\substack{I^{e}, x_{1}^{e}}}
    &= \frac{\partial}{\partial x_1} \left ( \frac{I^2}{ \left( \delta - x_1 \right)^2} \right ) \Bigr|_{\substack{I^{e}, x_{1}^{e}}}
    = I^2 \frac{\partial}{\partial x_1} \left ( \left( \delta - x_1 \right)^{-2} \right ) \Bigr|_{\substack{I^{e}, x_{1}^{e}}} \cr
    &= \frac{-2 I^2}{ \left (\delta - x_1 \right)^3} \frac{\mathrm{d}}{\mathrm{d} x_1} \left( - x_1 \right) \Bigr|_{\substack{I^{e}, x_{1}^{e}}}
    = \frac{2 I^2}{ \left (\delta - x_1 \right)^3} \Bigr|_{\substack{I^{e}, x_{1}^{e}}}
    = \frac{2 \left(I^{e} \right)^{2}}{ \left (\delta - x_1^e \right)^3}
\end{align}
\noindent Substitute equations \eqref{a4:partial_psi_i} and \eqref{a4:partial_psi_x1} into equation \eqref{a4:taylor_eqn_phi}:
\begin{equation} \label{a4:taylor_of_phi}
    \psi \left( I, x_{1} \right)
    \approx \frac{ \left(I^{e} \right)^2}{ \left( \delta - x_{1}^e \right)^2 } + \frac{2 I^e}{ \left( \delta - x_{1}^e \right)^2} \left( I - I^e \right) + \frac{2  \left(I^{e} \right)^2}{ \left (\delta - x_{1}^e \right)^3} (x_{1} - x_{1}^{e} )
\end{equation}
\noindent Linearise $\lambda$: 
\begin{equation} \label{a4:taylor_eqn_lambda}
    \lambda \left( \beta, x_{1} \right) \approx 
    \lambda \left( \beta^{e}, x_{1}^{e} \right)
    + \frac{\partial \lambda}{\partial \beta}
    \Bigr|_{\substack{\beta^{e}, x_{1}^{e}}}
     \left( \beta - \beta^e \right)  
    + \frac{\partial \lambda}{\partial x_1}
    \Bigr|_{\substack{\beta^{e}, x_{1}^{e}}} (x_{1} - x_{1}^{e} ) 
\end{equation}
\noindent Find the partial derivatives:
\begin{align}
    \frac{\partial \lambda}{\partial \beta} \Bigr|_{\substack{\beta^{e}, x_{1}^{e}}}
    &= \frac{\partial}{\partial \beta}\left( \frac{\zeta \beta} { L_0 + L_1 e^{- \alpha \left( \delta - x_{1} \right)}} \right) \Bigr|_{\substack{\beta^{e}, x_{1}^{e}}}
    = \frac{\zeta} { L_0 + L_1 e^{- \alpha \left( \delta - x_{1} \right)}} \Bigr|_{\substack{\beta^{e}, x_{1}^{e}}}
    = \frac{\zeta} { L_0 + L_1 e^{- \alpha \left( \delta - x_{1}^{e} \right)}} \label{a4:partial_lambda_beta} \\ \vspace{2mm}
    \frac{\partial \lambda}{\partial x_1} \Bigr|_{\substack{\beta^{e}, x_{1}^{e}}}
    &= \frac{\partial}{\partial x_1}\left( \frac{\zeta \beta} { L_0 + L_1 e^{- \alpha \left( \delta - x_{1} \right)}} \right) \Bigr|_{\substack{ \beta^{e}, x_{1}^{e}}} \cr
    &= \zeta \beta \left( - 1 \right) \left( L_0 + L_1 e^{- \alpha \left( \delta - x_{1} \right)} \right)^{-2}
        \frac{\partial}{\partial x_1} \left( L_0 + L_1 e^{- \alpha \left( \delta - x_{1} \right)} \right) \Bigr|_{\substack{  \beta^{e}, x_{1}^{e}}} \cr
    &= \zeta \beta \left( - 1 \right) \left( L_0 + L_1 e^{- \alpha \left( \delta - x_{1} \right)} \right)^{-2} \left( L_1 e^{- \alpha \left( \delta - x_{1} \right)} \right)
        \frac{\partial}{\partial x_1} \left( - \alpha \left( \delta - x_{1} \right) \right) \Bigr|_{\substack{  \beta^{e}, x_{1}^{e}}  } \cr
    &= \zeta \beta \left( - 1 \right) \left( L_0 + L_1 e^{- \alpha \left( \delta - x_{1} \right)} \right)^{-2} \left( L_1 e^{- \alpha \left( \delta - x_{1} \right)} \right) \left( \alpha \right) \Bigr|_{\substack{  \beta^{e}, x_{1}^{e}}} \cr
    &= \frac{- \alpha \zeta \beta^e \left( L_1 e^{- \alpha \left(  \delta - x_{1}^e \right)} \right)}{\left( L_0 + L_1 e^{- \alpha \left( \delta - x_{1}^e \right)} \right)^{2}} \label{a4:partial_lambsa_x1}
\end{align}
\noindent Substitute equations \eqref{a4:partial_lambda_beta} and \eqref{a4:partial_lambsa_x1} into equation \eqref{a4:taylor_eqn_lambda}:
\begin{equation}
    \lambda \left( \beta, x_{1} \right)  
    \approx
    \frac{\zeta \beta^e}{ L_0 + L_1 e^{- \alpha \left( \delta - x_{1}^e \right)}}
    + \frac{\zeta}{ L_0 + L_1 e^{- \alpha \left( \delta - x_{1}^e \right)}} (\beta - \beta^e )
    - \frac{ \alpha \zeta \beta^e L_1 e^{- \alpha (\delta - x_{1}^e)}}{ \left( L_0 + L_1 e^{- \alpha \left( \delta - x_{1}^e \right)} \right)^2} (x_{1} - x_{1}^{e} )
\end{equation}
\begin{flalign}
    \text{Let } &N = \frac{1}{ L_0 + L_1 e^{- \alpha \left( \delta - x_{1}^e \right)}} \label{constant:N} \\
    \Longrightarrow &\lambda \left( \beta, x_{1} \right)  
    \approx
    \zeta \beta^e N
    + \zeta N (\beta - \beta^e )
    - \alpha \zeta \beta^e L_1 e^{- \alpha (\delta - x_{1}^e)} N^2 (x_{1} - x_{1}^{e} ) \label{a4:tayler_of_lambda}&&
\end{flalign}
\noindent The linearisation of $\dot{I}$ can be found by using equation \eqref{a4:tayler_of_lambda}. By substituting $\zeta$ as $1$ and $\beta$ as $V$, as well as introducing the deviation variables, $\frac{V}{L_0 + L_1 e^{- \alpha \left( \delta - x_1 \right)}}$ can be approximated as:
\begin{equation} \label{a4:approx1}
    \frac{V}{L_0 + L_1 e^{- \alpha \left( \delta - x_1 \right)}}
    \approx
    V^e N
    + N \bar{V}
    - V^e L_1 \alpha e^{- \alpha (\delta - x_{1}^e)} N^2 \bar{x}_1
\end{equation}
\noindent By substituting $\zeta$ as $R$ and $\beta$ as $I$, as well as introducing the deviation variables, $\frac{IR}{L_0 + L_1 e^{- \alpha \left( \delta - x_1 \right)}}$ can be approximated as:
\begin{equation} \label{a4:approx2}
    \frac{IR}{L_0 + L_1 e^{- \alpha \left( \delta - x_1 \right)}}
    \approx
    I^e R N
    + R N \bar{I}
    - I^e R L_1 \alpha e^{- \alpha (\delta - x_{1}^e)} N^2 \bar{x}_1
\end{equation}
\noindent Subtract equation \eqref{a4:approx2} from equation \eqref{a4:approx1}:
\begin{equation} \label{a4:current_dot_0}
    \frac{V - IR}{L_0 + L_1 e^{- \alpha \left( \delta - x_1 \right)}}
    \approx
    \left( V^e - I^e R \right) N
    + N \bar{V}
    - R N \bar{I}
    - \left( V^e - I^e R \right) L_1 \alpha e^{- \alpha (\delta - x_{1}^e)} N^2 \bar{x}_1
\end{equation}
\noindent Substitute equation \eqref{a3:equil_current} into \eqref{a4:current_dot_0}:
\begin{equation} \label{a4:current_dot_1}
    \dot{\bar{I}} = N \bar{V} - R N \bar{I}
\end{equation}
\noindent Substitute equations \eqref{a4:taylor_of_phi} and \eqref{a4:current_dot_1}, as well as introducing the deviation variables into \eqref{a4:pre-linear} gives:
\begin{align} \label{a4:linear}
    \underbar{$\dot{\bar{x}}$}
    &=
    \begin{bmatrix}
        \bar{x}_2 \cr
        \left( \frac{5}{7m} \right)
        \left( \frac{2c I^e}{ \left( \delta - x_{1}^e \right)^2} \bar{I}
        + \left( \frac{ 2c \left(I^{e} \right)^2}{ \left (\delta - x_{1}^e \right)^3} - k \right) \bar{x}_1
        - b \bar{x}_2 \right) \cr
        N \bar{V}
        - R N \bar{I}
    \end{bmatrix}
\end{align}
\noindent The following constants are for notational convenience:
\begin{align}
    D &= \left( \frac{5}{7m} \right) \left( \frac{2c I^e}{ \left( \delta - x_{1}^e \right)^2} \right) \label{constant:D}\\
    F &= \left( \frac{5}{7m} \right) \left( \frac{ 2c \left(I^{e} \right)^2}{ \left (\delta - x_{1}^e \right)^3} - k \right) \label{constant:F}\\
    H &= \left( \frac{5}{7m} \right) \left( b \right) \label{constant:H}\\
    P &= RN = \frac{R}{ L_0 + L_1 e^{- \alpha \left( \delta - x_{1}^e \right)}} \label{constant:P}
\end{align}
\noindent To get the linearised system, substitute equations \eqref{constant:D} to \eqref{constant:P} into equation \eqref{a4:linear}:
\begin{equation} \label{a4:linear_system}
    \underbar{$\dot{\bar{x}}$}
    =
    \begin{bmatrix}
        \bar{x}_2 \cr
        D \bar{I} + F \bar{x}_1 - H \bar{x}_2 \cr
        N \bar{V} - P \bar{I} \cr
    \end{bmatrix}
\end{equation}
