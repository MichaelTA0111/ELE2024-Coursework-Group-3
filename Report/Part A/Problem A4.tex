\bigskip
\subsection{Problem A4}\label{sec:qa4}\hfill

\medskip
Linearise $f \left( \underbar{$\dot{\bar{x}}$}, \bar{V} \right) $ at $\underbar{$x$}^e$,$V^{e}$ as time $t\to \infty$

\bigskip
To Linearise $f$, $\psi = \frac{{ \left( \frac{V}{R} 
- \frac{V}{R}e^{\frac{-R}{L}t} \right) }^2}{{ \left( \delta - x_{1} \right) }^2} $ needs to be Linearised.

Taylors Theorm is used for Linearisation. It is modified a little to account for $t \to \infty$ .
\begin{equation}
\psi \left( x_{1},V \right)  \approx 
 \lim_{t\to\infty}
\left(
\psi \left( x_{1}^{e},V^{e} \right) 
+ \frac{\partial \psi}{\partial x_{1}}
%\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) }
\Bigr|_{\substack{ ( x_{1}^{e},V^{e}} ) }
 \left( x_{1} - x_{1}^e \right)  
+ \frac{\partial \psi}{\partial V}
%\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) } \left( V - V^{e} \right) \]
\Bigr|_{\substack{ ( x_{1}^{e},V^{e} ) }} ( V - V^{e} ) 
\right)
\end{equation}

\begin{equation}
\psi \left( x_{1},V \right)  \approx 
 \lim_{t\to\infty}
\psi \left( x_{1}^{e},V^{e} \right) 
+  \lim_{t\to\infty}
\frac{\partial \psi}{\partial x_{1}}
%\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) }
\Bigr|_{\substack{ ( x_{1}^{e},V^{e}} ) }
 \left( x_{1} - x_{1}^e \right)  
+  \lim_{t\to\infty}
\frac{\partial \psi}{\partial V}
%\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) } \left( V - V^{e} \right) \]
\Bigr|_{\substack{ ( x_{1}^{e},V^{e} ) }} ( V - V^{e} ) 
\end{equation}
 
 Find $\psi \left( x_{1},V \right)$.
 \begin{align*}
\lim_{t\to\infty}
 \psi \left( x_{1},V \right)  
 &=
\lim_{t\to\infty}
\frac{{ \left( \frac{V^{e}}{R} - \frac{V^{e}}{R}e^{\frac{-R}{L}t} \right) }^2}{{ \left( \delta - x_{1}^e \right) }^2}
\cr
&=
\frac{{ \lim_{t\to\infty}
\left( \frac{V^{e}}{R} - \frac{V^{e}}{R}e^{\frac{-R}{L}t} \right) }^2}{{ \left( \delta - x_{1}^e \right) }^2}
\cr
&=
\frac{{ \left( \frac{V^{e}}{R} - \frac{V^{e}}{R} \times 0 \right) }^2}{{ \left( \delta - x_{1}^e \right) }^2}
\cr
&=
\frac{{ \left( \frac{V^{e}}{R} \right) }^2}{{ \left( \delta - x_{1}^e \right) }^2}
\end{align*}

Find Each Derivative
\begin{align*}
\lim_{t\to\infty} \left(
\frac{\partial }{\partial x_{1}}
{ \left( \frac{V}{R} - \frac{V}{R}e^{\frac{-R}{L}t} \right) }^2
{ \left( \delta - x_{1} \right) }^{-2}
%\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) }
\Bigr|_{\substack{ ( x_{1}^{e},V^{e}} ) }
\right)
&=
\lim_{t\to\infty} \left(
{ \left( \frac{V}{R} - \frac{V}{R}e^{\frac{-R}{L}t} \right) }^2
\frac{\partial }{\partial x_{1}}
{ \left( \delta - x_{1} \right) }^{-2}
%\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) }
\Bigr|_{\substack{ ( x_{1}^{e},V^{e}} ) }
\right)
\cr
&=
\lim_{t\to\infty} \left(
{ \left( \frac{V}{R} - \frac{V}{R}e^{\frac{-R}{L}t} \right) }^2
\left( -2 \right) \left( -1 \right)
{ \left( \delta - x_{1} \right) }^{-3}
%\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) }
\Bigr|_{\substack{ ( x_{1}^{e},V^{e}} ) }
\right)
\cr
&=
\lim_{t\to\infty} \left(
\left( 2 \right)
\frac
{{ \left( \frac{V}{R} - \frac{V}{R}e^{\frac{-R}{L}t} \right) }^2}
{{ \left( \delta - x_{1} \right) }^{3}}
%\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) }
\Bigr|_{\substack{ ( x_{1}^{e},V^{e}} ) }
\right)
\cr
&=
\lim_{t\to\infty} \left(
\left( 2 \right)
\frac
{{ \left( \frac{V^{e}}{R} - \frac{V^{e}}{R}e^{\frac{-R}{L}t} \right) }^2}
{{ \left( \delta -  x_{1}^{e} \right) }^{3}}
\right)
\cr
&=
\left( 2 \right)
\frac
{\lim_{t\to\infty}
{ \left( \frac{V^{e}}{R} - \frac{V^{e}}{R}e^{\frac{-R}{L}t} \right) }^2}
{{ \left( \delta -  x_{1}^{e} \right) }^{3}}
\cr
&=
\left( 2 \right)
\frac{{ \left( \frac{V^{e}}{R} - \frac{V^{e}}{R} \times 0 \right) }^2}
{{ \left( \delta -  x_{1}^{e} \right) }^{3}}
\cr
&=
\left( 2 \right)
\frac{{ \left( \frac{V^{e}}{R} \right) }^2}
{{ \left( \delta -  x_{1}^{e} \right) }^{3}}
\end{align*}
 
\begin{align*}
\lim_{t\to\infty} \left(
\frac{\partial }{\partial V}
{ \left( \frac{V}{R} - \frac{V}{R}e^{\frac{-R}{L}t} \right) }^{2}
{ \left( \delta - x_{1} \right) }^{-2}
%\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) }
\Bigr|_{\substack{ ( x_{1}^{e},V^{e}} ) }
\right)
&=
\lim_{t\to\infty} \left(
{ \left( \delta - x_{1} \right) }^{-2}
\frac{\partial }{\partial V}
{ \left( \frac{V}{R} - \frac{V}{R}e^{\frac{-R}{L}t} \right) }^{2}
%\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) }
\Bigr|_{\substack{ ( x_{1}^{e},V^{e}} ) }
\right)
\cr
&=
\lim_{t\to\infty} \left(
{ \left( \delta - x_{1} \right) }^{-2}
\left( 2 \right)
{ \left( \frac{V}{R} - \frac{V}{R}e^{\frac{-R}{L}t} \right) }
{ \left( \frac{1}{R} - \frac{1}{R}e^{\frac{-R}{L}t} \right) }
%\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) }
\Bigr|_{\substack{ ( x_{1}^{e},V^{e}} ) }
\right)
\cr
&=
\lim_{t\to\infty} \left(
\left( 2 \right)
\frac
{{ \left( \frac{V}{R} - \frac{V}{R}e^{\frac{-R}{L}t} \right) }
{ \left( \frac{1}{R} - \frac{1}{R}e^{\frac{-R}{L}t} \right) }}
{{ \left( \delta - x_{1} \right) }^{2}}
%\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) }
\Bigr|_{\substack{ ( x_{1}^{e},V^{e}} ) }
\right)
\cr
&=
\left( 2 \right)
\frac
{\lim_{t\to\infty}
{ \left( \frac{V^{e}}{R} - \frac{V^{e}}{R}e^{\frac{-R}{L}t} \right) }
\lim_{t\to\infty}
{ \left( \frac{1}{R} - \frac{1}{R}e^{\frac{-R}{L}t} \right) }}
{{ \left( \delta - x_{1}^{e} \right) }^{2}}
\cr
&=
\left( 2 \right)
\frac{
{ \left( \frac{V^{e}}{R} - \frac{V^{e}}{R} \times 0 \right) }
{ \left( \frac{1}{R} - \frac{1}{R} \times 0 \right) }}
{{ \left( \delta - x_{1}^{e} \right) }^{2}}
\cr
&=
\left( 2 \right)
\frac{
{ \left( \frac{V^{e}}{R} \right) }
{ \left( \frac{1}{R} \right) }}
{{ \left( \delta - x_{1}^{e} \right) }^{2}}
\cr
&=
\left( 2 \right)
\frac{
{ \left( \frac{V^{e}}{R^2} \right) }}
{{ \left( \delta - x_{1}^{e} \right) }^{2}}
\end{align*}
