\end{align*}

\medskip
Linearise $f \left( \underbar{$\dot{\bar{x}}$}, \bar{V} \right)$, at  $\underbar{$x$}^e$, $I^{e}$, $V^{e}$.

\bigskip
To linearise $f$, $\psi = \frac{I^2}{ \left( \delta - x_1 \right)^2} $ and $\lambda =\frac{\beta \zeta}{L_0 + L_1 e^{- \alpha \left( \delta - x_1 \right)}}$ needs to be linearised.\\

NOTE: $\beta$ and $\zeta$ represent a variable, and a constant respectively.

Taylor's Theorem is used for linearisation. \\

\bigskip
Linearise $\psi$:
\begin{equation}
    \psi \left( I, x_{1} \right)  \approx 
    \psi \left( I^{e}, x_{1}^{e} \right)
    + \frac{\partial \psi}{\partial I}
    %\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) }
    \Bigr|_{\substack{ ( I^{e}, x_{1}^{e}} ) }
     \left( I - I^e \right)  
    + \frac{\partial \psi}{\partial x_1}
    %\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) } \left( V - V^{e} \right) \]
    \Bigr|_{\substack{ ( I^{e}, x_{1}^{e}) }} (x_{1} - x_{1}^{e} ) 
\end{equation}
 
\bigskip
Find the partial derivatives:
\begin{align*}
    \frac{\partial \psi}{\partial I}
    \Bigr|_{\substack{ ( I^{e}, x_{1}^{e}} )} 
    &=
    \frac{d}{d I} \left ( \frac{I^2}{ \left( \delta - x_1 \right)^2} \right ) 
    \Bigr|_{\substack{ ( I^{e}, x_{1}^{e}} )} 
    \cr 
    &=
    \frac{2 I}{ \left( \delta - x_1 \right)^2}
    \Bigr|_{\substack{ ( I^{e}, x_{1}^{e}} )} 
    \cr 
    &=
    \frac{2 I^e}{ \left( \delta - x_1^e \right)^2}
\end{align*}

\begin{align*}
    \frac{\partial \psi}{\partial x_1}
    \Bigr|_{\substack{ ( I^{e}, x_{1}^{e}} )}
    &=
    \frac{d}{d x_1} \left (I^2 \frac{1}{ \left( \delta - x_1 \right)^2} \right ) 
    \Bigr|_{\substack{ ( I^{e}, x_{1}^{e}} )}
    \cr
    &=
    I^2 \frac{d}{d x_1} \left ( \left( \delta - x_1 \right)^{-2} \right ) 
    \Bigr|_{\substack{ ( I^{e}, x_{1}^{e}} )}
    \cr 
    &=
    \frac{-2 I^2}{ \left (\delta - x_1 \right)^3} \times
    \frac{d}{d x_1} \left( - x_1 \right)
    \Bigr|_{\substack{ ( I^{e}, x_{1}^{e}} )}
    \cr
    &=
    \frac{2 I^2}{ \left (\delta - x_1 \right)^3}
    \Bigr|_{\substack{ ( I^{e}, x_{1}^{e}} )}
    \cr
    &=
    \frac{2 \left(I^{e} \right)^{2}}{ \left (\delta - x_1^e \right)^3}
    \cr
\end{align*}

\begin{align}
    \psi \left( I, x_{1} \right)  &\approx 
    \frac{ \left(I^{e} \right)^2}{ \left( \delta - x_{1}^e \right)^2 }
    + \frac{2 I^e}{ \left( \delta - x_{1}^e \right)^2}
     \left( I - I^e \right)  
    + \frac{2  \left(I^{e} \right)^2}{ \left (\delta - x_{1}^e \right)^3} 
    (x_{1} - x_{1}^{e} ) \cr
    &\approx \frac{ \left(I^{e} \right)^2}{ \left( \delta - x_{1}^e \right)^2 }
    + \frac{2 I^e}{ \left( \delta - x_{1}^e \right)^2} \bar{I}  
    + \frac{2  \left(I^{e} \right)^2}{ \left (\delta - x_{1}^e \right)^3} \bar{x_{1}}
\end{align}

\bigskip 
\noindent Linearise $\lambda$: 
\begin{equation}
    \lambda \left( \beta, x_{1} \right)  \approx 
    \lambda \left( \beta^{e}, x_{1}^{e} \right)
    + \frac{\partial \lambda}{\partial \beta}
    \Bigr|_{\substack{ ( \beta^{e}, x_{1}^{e}} ) }
     \left( \beta - \beta^e \right)  
    + \frac{\partial \lambda}{\partial x_1}
    \Bigr|_{\substack{ ( \beta^{e}, x_{1}^{e}) }} (x_{1} - x_{1}^{e} ) 
\end{equation}

\bigskip
Find the partial derivatives:
\begin{align*}
    \frac{\partial \lambda}{\partial \beta}
    \Bigr|_{\substack{ ( \beta^{e}, x_{1}^{e}} ) }
    &=
    \frac{\partial}{\partial \beta}\left(
    \frac{\beta \zeta} { L_0 + L_1 e^{- \alpha \left( \delta - x_{1} \right)}} \right)
    \Bigr|_{\substack{ ( \beta^{e}, x_{1}^{e}} ) }
    \cr
    &=
    \frac{\zeta} { L_0 + L_1 e^{- \alpha \left( \delta - x_{1} \right)}}
    \Bigr|_{\substack{ ( \beta^{e}, x_{1}^{e}} ) }
    \cr
    &=
    \frac{\zeta} { L_0 + L_1 e^{- \alpha \left( \delta - x_{1}^{e} \right)}}
    \cr
\end{align*}

\begin{align*}
    \frac{\partial \lambda}{\partial x_1}
    \Bigr|_{\substack{ ( \beta^{e}, x_{1}^{e}} )} 
    &=
    \frac{\partial}{\partial x_1}\left(
    \frac{\beta \zeta} { L_0 + L_1 e^{- \alpha \left( \delta - x_{1} \right)}} \right)
    \Bigr|_{\substack{ ( \beta^{e}, x_{1}^{e}} ) }
    \cr
    &=
    \beta \zeta 
    \frac{\partial}{\partial x_1}\left(
    \left( L_0 + L_1 e^{- \alpha \left( \delta - x_{1} \right)}\right)^{-1} \right)
    \Bigr|_{\substack{ ( \beta^{e}, x_{1}^{e}} ) }
    \cr
    &=
    \beta \zeta \left( - 1 \right)
    \left( 
            L_0 + L_1 e^{- \alpha \left( \delta - x_{1} \right)}
    \right)^{-2}
    \frac{\partial}{\partial x_1}
    \left(
        L_0 + L_1 e^{- \alpha \left( \delta - x_{1} \right)}
    \right)
    \Bigr|_{\substack{ ( \beta^{e}, x_{1}^{e}} ) }
    \cr
    &=
    \beta \zeta \left( - 1 \right)
    \left( 
            L_0 + L_1 e^{- \alpha \left( \delta - x_{1} \right)}
    \right)^{-2}
    \left(
        L_1 e^{- \alpha \left( \delta - x_{1} \right)}
    \right)
    \frac{\partial}{\partial x_1}
    \left(
        - \alpha x_{1}
    \right)
    \Bigr|_{\substack{ ( \beta^{e}, x_{1}^{e}} ) }
    \cr
    &=
    \beta \zeta \left( - 1 \right)
    \left( 
            L_0 + L_1 e^{- \alpha \left( \delta - x_{1} \right)}
    \right)^{-2}
    \left(
        L_1 e^{- \alpha \left( \delta - x_{1} \right)}
    \right)
    \left(
        - \alpha
    \right)
    \Bigr|_{\substack{ ( \beta^{e}, x_{1}^{e}} ) }
    \cr
    &= 
    -\beta \zeta \left( L_0 + L_1 e^{- \alpha \left( \delta - x_{1} \right)} \right)^{-2} \left( L_1 - \alpha e^{- \alpha \left(  \delta - x_1 \right)} \right) 
    \Bigr|_{\substack{ ( \beta^{e}, x_{1}^{e}} ) }
    \cr
    &=
    \frac{- \beta \zeta \left( L_1 - \alpha e^{- \alpha \left(  \delta - x_1 \right)} \right)}{\left( L_0 + L_1 e^{- \alpha \left( \delta - x_{1} \right)} \right)^{2}} 
    \Bigr|_{\substack{ ( \beta^{e}, x_{1}^{e}} )}
    \cr
    &= 
    \frac{- \beta^e \zeta \left( L_1 - \alpha e^{- \alpha \left(  \delta - x_{1}^e \right)} \right)}{\left( L_0 + L_1 e^{- \alpha \left( \delta - x_{1}^e \right)} \right)^{2}}
    \cr
\end{align*}

\begin{align}
    \lambda \left( \beta, x_{1} \right)  &\approx 
    \frac{\beta^e \zeta}{ L_0 + L_1 e^{- \alpha \left( \delta - x_{1}^e \right)}}
    + \frac{\zeta}{ L_0 + L_1 e^{- \alpha \left( \delta - x_{1}^e \right)}} (\beta - \beta^e )
    + \frac{- \beta^e \zeta L_1 - \alpha e^{- \alpha (\delta - x_{1}^e)}}{ \left( L_0 + L_1 e^{- \alpha \left( \delta - x_{1}^e \right)} \right)^2} (x_{1} - x_{1}^{e} ) \cr
    &\approx 
    \frac{\beta^e \zeta}{ L_0 + L_1 e^{- \alpha \left( \delta - x_{1}^e \right)}}
    + \frac{\zeta}{ L_0 + L_1 e^{- \alpha \left( \delta - x_{1}^e \right)}} \bar{\beta}
    + \frac{- \beta^e \zeta L_1 - \alpha e^{- \alpha (\delta - x_{1}^e)}}{ \left( L_0 + L_1 e^{- \alpha \left( \delta - x_{1}^e \right)} \right)^2} \bar{x_1}
\end{align}

\bigskip \noindent
Using (18), the linearisation of $\dot{I}$ can be found.\\
By substituting $\beta$ as $V$, and $\zeta$ as $1$, $\frac{V}{L_0 + L_1 e^{- \alpha \left( \delta - x_1 \right)}}$ can be found.
\begin{equation}
    \frac{V}{L_0 + L_1 e^{- \alpha \left( \delta - x_1 \right)}}
    \approx
    \frac{V^e}{ L_0 + L_1 e^{- \alpha \left( \delta - x_{1}^e \right)}}
    + \frac{1}{ L_0 + L_1 e^{- \alpha \left( \delta - x_{1}^e \right)}} \bar{V}
    + \frac{- V^e L_1 - \alpha e^{- \alpha (\delta - x_{1}^e)}}{ \left( L_0 + L_1 e^{- \alpha \left( \delta - x_{1}^e \right)} \right)^2} \bar{x_1}
\end{equation}

\medskip \noindent
By substituting $\beta$ as $I$, and $\zeta$ as $R$, $\frac{IR}{L_0 + L_1 e^{- \alpha \left( \delta - x_1 \right)}}$ can be found.
\begin{equation}
    \frac{V}{L_0 + L_1 e^{- \alpha \left( \delta - x_1 \right)}}
    \approx
    \frac{I^e R}{ L_0 + L_1 e^{- \alpha \left( \delta - x_{1}^e \right)}}
    + \frac{R}{ L_0 + L_1 e^{- \alpha \left( \delta - x_{1}^e \right)}} \bar{I}
    + \frac{- I^e R L_1 - \alpha e^{- \alpha (\delta - x_{1}^e)}}{ \left( L_0 + L_1 e^{- \alpha \left( \delta - x_{1}^e \right)} \right)^2} \bar{x_1}
\end{equation}
