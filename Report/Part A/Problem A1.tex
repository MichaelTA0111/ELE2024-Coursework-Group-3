\subsection{Problem A1}\label{sec:qa1}

\begin{equation}
    m\ddot{x} =
    mg\sin \left( \phi \right)
    + F_{magnet}
    - k \left( x - d \right)
    - b \dot{x}
    + F_{friction}
\end{equation}

\begin{equation}
    F_{magnet} = c \frac{I^2}{y^2}
\end{equation}

\noindent We need to find equations for $I$ and $y$.

\begin{equation}
    y = \delta - x
\end{equation}

\noindent Equation for $y$ found.

\begin{align*}
    V &= IR + \dot{I}L\\
    V &= IR + \frac{dI}{dt}L\\
    V - IR &= \frac{dI}{dt}L\\
    \frac{V - IR}{dI} &= \frac{L}{dt}\\
    \frac{1}{V - IR}dI &= \frac{1}{L}dt\\
    \int \frac{1}{V - IR}dI &= \int \frac{1}{L}dt
\end{align*}

\noindent \\Let $u = V - IR$

\begin{align*}
    \Longrightarrow \frac{du}{dI} &= -R\\
    dI &= \frac{1}{-R} du\\
   \int_{0}^{u} \frac{1}{u} \times \frac{1}{-R} du &= \int_{0}^{t} \frac{1}{L} dt\\
   \frac{1}{-R} \int_{0}^{u} \frac{1}{u} du &= \int_{0}^{t} \frac{1}{L} dt\\
   \frac{1}{-R} \left[ \ln \left( u \right) \right]_{0}^{u} &= \frac{1}{L}t\\
   \frac{1}{-R} \left[ \ln \left( V - IR \right) \right]_{0}^{I}  &= \frac{1}{L}t\\
   \left[ \ln \left( V - IR \right) \right]_{0}^{I} &= \frac{-R}{L}t\\
   \ln \left( V - IR \right)  - \ln \left( V \right) &= \frac{-R}{L}t\\
   \ln \left( \frac{V - IR}{V} \right) &= \frac{-R}{L}t\\
   \frac{V - IR}{V} &= e^{\frac{-R}{L}t}\\
   V - IR &= Ve^{\frac{-R}{L}t}\\
   V - Ve^{\frac{-R}{L}t} &= IR\\
   I &= \frac{V}{R} - \frac{V}{R}e^{\frac{-R}{L}t}\\
\end{align*}

\noindent Equation for $I$ found.

\noindent Substituting $I$ and $y$ into $F_{magnet}$.

\bigskip
\begin{equation}
    F_{magnet} = c \frac{{ \left( \frac{V}{R} - \frac{V}{R}e^{\frac{-R}{L}t} \right) }^2}{{ \left( \delta - x \right) }^2}
\end{equation}

\noindent Now we need to find an equation for $F_{friction}$

\bigskip
\begin{equation*}
    -F_{friction} \times r = - I_{inertia} \ddot{\theta}
\end{equation*}

\noindent ``Pizza Slice Theorem'':

\begin{equation*}
    \ddot{x} = \ddot{\theta} r
\end{equation*}

\noindent Inertia of a solid sphere:

\begin{align}
    I_{inertia} &= \frac{2}{5}mr^2\notag\\
    F_{friction} \times r &= \frac{2}{5}mr^2 \times \frac{\ddot{x}}{r}\notag\\
    F_{friction} \times r &= \frac{2}{5}mr \ddot{x}\notag\\
    F_{friction} &= \frac{2}{5}m \ddot{x}
\end{align}

\noindent Equation for $F_{friction}$ found.

\bigskip
\noindent Substituting (2), (3), (4) and (5) into (1).
\begin{equation}
    m\ddot{x} =
    mg\sin \left( \phi \right)
    + c \frac{{ \left( \frac{V}{R} \left( 1 - e^{\frac{-R}{L}t}\right) \right) }^2}{{ \left( \delta - x \right) }^2}
    - k \left( x - d \right)
    - b \dot{x}
    + \frac{2}{5}m \ddot{x}
\end{equation}

\bigskip
\noindent Simplify (6):
\begin{align}
    m\ddot{x} - \frac{2}{5}m \ddot{x} &=
    mg\sin \left( \phi \right)
    + c \frac{{ \left( \frac{V}{R} \left( 1 - e^{\frac{-R}{L}t}\right) \right) }^2}{{ \left( \delta - x \right) }^2}
    - k \left( x - d \right)
    - b \dot{x}\notag \\
    \left( 1 - \frac{2}{5} \right) m \ddot{x} &=
    mg\sin \left( \phi \right)
    + c \frac{{ \left( \frac{V}{R} \right)^2 \left( 1 - e^{\frac{-R}{L}t}\right) }^2}{{ \left( \delta - x \right) }^2}
    - k \left( x - d \right)
    - b \dot{x}\notag \\
    \frac{3m}{5} \ddot{x} &=
    mg\sin \left( \phi \right)
    + c \frac{{ V^2 \left( 1 - e^{\frac{-R}{L}t}\right) }^2}{R^2{ \left( \delta - x \right) }^2}
    - k \left( x - d \right)
    - b \dot{x}\notag \\
    \ddot{x} &=
    \left( \frac{5}{3m} \right)
    \left( mg\sin \left( \phi \right)
    + c \left( \frac{{ V \left( 1 - e^{\frac{-R}{L}t}\right) }}{{ R \left( \delta - x \right) }} \right)^2
    - k \left( x - d \right)
    - b \dot{x} \right)
\end{align}
