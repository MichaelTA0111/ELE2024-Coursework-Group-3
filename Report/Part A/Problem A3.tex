\bigskip
\subsection{Problem A3}\label{sec:qa3}\hfill

\medskip
\noindent Proposition 3.4 in the textbook states that if $\dot{x} = f \left( x \left( t \right) , u \left( t \right)  \right)$,
\noindent $\dot{x} = 0 = f \left( x^e, u^e \right) $ for all $t \geqslant 0$.\\
\noindent The equilibrium points are
$
    \underbar{$x$}^e = 
    \begin{bmatrix}
    x_{1}^e \cr
    x_{2}^e \cr
    I^e \cr
    \end{bmatrix}
$
, $V^e$.

\medskip
\noindent Substituting in the equilibrium values and using Proposition 3.4 produces the following:
\begin{equation} \label{a3:equilibrium_0}
    0
    = 
    \begin{bmatrix}
        x_{2}^e \cr
        \left( \frac{5}{7m} \right)
        \left( mg\sin \phi
        + c \frac{ \left(I^{e} \right)^2}{ \left( \delta - x_{1}^e \right)^2 }
        - k \left( x_{1}^e - d \right)
        - b x_{2}^e \right) \cr
        \frac{1}{L_0 + L_1 e^{-\alpha \left( \delta - x_{1}^e \right)}} \left(V^e - I^e R \right)
    \end{bmatrix}
\end{equation}
\begin{flalign} \label{a3:equil_current}
    \text{Substituting the equilibrium values into equation \eqref{a1:current_dot} gives } 0 &= V^e - I^e R&&
\end{flalign}
\noindent Substituting equation \eqref{a3:equil_current} into equation \eqref{a3:equilibrium_0} gives:
\begin{equation} \label{a3:equilibrium_1}
    0
    = 
    \begin{bmatrix}
        x_{2}^e \cr
        \left( \frac{5}{7m} \right)
        \left( mg\sin \phi
        + c \frac{ \left(I^{e} \right)^2}{ \left( \delta - x_{1}^e \right)^2 }
        - k \left( x_{1}^e - d \right)
        - b x_{2}^e \right) \cr
        0
    \end{bmatrix}
\end{equation}
