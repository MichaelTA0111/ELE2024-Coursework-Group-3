\subsection{Problem A5}\label{sec:qa5}\hfill

\medskip
\noindent Apply Laplace Transform to equation (30), with all initial conditions equalling zero:

\begin{align}
    \lap \{\dot{\bar{x}}_1\} &= \lap \{\bar{x}_2\} \cr
    s \bar{X}_1 - \bar{x}_1 (0^+) &= \bar{X}_2 \cr
    s \bar{X}_1 &= \bar{X}_2
\end{align}

\begin{align}
    \lap \{ \dot{\bar{x}}_2 \} &= \lap \{D \bar{I} + F \bar{x}_1 - H \bar{x}_2 \} \cr
    \lap \{ \dot{\bar{x}}_2 \} &= D \lap \{ \bar{I} \} + F \lap \{\bar{x}_1 \} - H \lap \{ \bar{x}_2 \} \cr
    s \bar{X}_2 - \bar{x}_2 (0^+) &= D \bar{I} + F \bar{X}_1 - H \bar{X}_2 \cr
    s \bar{X}_2 &= D \bar{I} + F \bar{X}_1 - H \bar{X}_2
\end{align}

\begin{align}
    \lap \{ \dot{\bar{I}} \} &= \lap \{ N \bar{V} - P \bar{I} - Q \bar{x}_1 \} \cr
    \lap \{ \dot{\bar{I}} \} &= N \lap \{ \bar{V} \} - P \lap \{\bar{I} \} - Q \lap \{ \bar{x}_1 \} \cr
    s \bar{I} - \bar{I} (0^+) &= N \bar{V} - P \bar{I} - Q \bar{X}_1 \cr
    s \bar{I} &= N \bar{V} - P \bar{I} - Q \bar{X}_1
\end{align}

\medskip
\noindent Find the transfer function

\noindent Rearrange equation (33): 
\begin{align}
    s \bar{I} &= N \bar{V} - P \bar{I} - Q \bar{X}_1 \cr
    s \bar{I} + P \bar{I} &= N \bar{V} - Q \bar{X}_1 \cr
    \bar{I} \left( s + P \right) &= N \bar{V} - Q \bar{X}_1 \cr
    \bar{I} &= \frac{N \bar{V} - Q \bar{X}_1}{\left( s + P \right)}
\end{align}

\noindent Substitute equation (31) into (32): 
\begin{align}
    s \bar{X}_2 &= D \bar{I} + F \bar{X}_1 - H \bar{X}_2 \cr
    s^2 \bar{X}_1 &= D \bar{I} + F \bar{X}_1 - H s \bar{X}_1 \cr
    D \bar{I} &= s^2 \bar{X}_1 + H s \bar{X}_1 - F \bar{X}_1 \cr
    \bar{I} &= \frac{s^2 \bar{X}_1 + H s \bar{X}_1}{D} - F \bar{X}_1 \cr
    \bar{I} &= \bar{X}_1 \frac{ s^2 + H s - F }{D}
\end{align}

\noindent Equation (34) equals equation (35): 
\begin{align}
    \bar{I} &= \bar{I} \cr
    \frac{N \bar{V} - Q \bar{X}_1}{\left( s + P \right)} &= \bar{X}_1 \frac{ s^2 + H s - F }{D} \cr
    D N \bar{V} - D Q \bar{X}_1 &= \bar{X}_1 \left( s^2 + H s - F \right) \left( s + P \right) \cr
    D N \bar{V} &= \bar{X}_1 \left( \left( s^2 + H s - F \right) \left( s + P \right) + D Q \right) \cr
    D N \bar{V} &= \bar{X}_1 \left( s^3 + \left( H + P \right) s^2 + \left( HP - F \right) + DQ - FP \right) \cr
    \frac{\bar{X}_1}{\bar{V}} &= \frac{D N}{s^3 + \left( H + P \right) s^2 + \left( HP - F \right) + DQ - FP} \cr
\end{align}

\noindent Equation (36) shows the transfer function of the system. There are 3 poles in the system.
