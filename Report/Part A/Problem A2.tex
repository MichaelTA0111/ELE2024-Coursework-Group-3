\subsection{Problem A2}\label{sec:qa2}\hfill

\medskip
\noindent Equation (11) is a 2\textsuperscript{nd} order ODE $\therefore$ introduce 2 new variables:
\begin{align*}
    x_1 &= x\\
    x_2 &= \dot{x}_1\\
    \dot{x}_2 &= \ddot{x}_1\\
    \Longrightarrow \underbar{$x$} &= 
    \begin{bmatrix}
        x_1 \cr
        x_2 \cr
        I \cr
    \end{bmatrix}\\
    \underbar{$\dot{x}$} &= 
    \begin{bmatrix}
        \dot{x}_1 \cr
        \dot{x}_2 \cr
        \dot{I} \cr
    \end{bmatrix}
\end{align*}

\begin{equation}
    \underbar{$\dot{x}$} =
    \begin{bmatrix}
        x_2 \cr
        \left( \frac{5}{3m} \right)
        \left( mg\sin \left( \phi \right)
        + c \frac{I^2}{ \left( \delta - x_1 \right)^2 }
        - k \left( x_1 - d \right)
        - b x_2 \right)\cr
        \frac{1}{L_0 + L_1 e^{-\alpha \left( \delta - x_1 \right)}} \left(V - I R \right)
    \end{bmatrix}
\end{equation}

\medskip
\noindent Equation (12) is the state space representation of the system.

\bigskip
\noindent The states of the system are
$
    \underbar{$x$} = 
    \begin{bmatrix}
    x_1 \cr
    x_2 \cr
    I \cr
    \end{bmatrix}
$.\\
The input to the system is $V$.
