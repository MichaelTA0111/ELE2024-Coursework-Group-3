\subsection{Problem A2}\label{sec:qa2}\hfill

\medskip
\noindent State space representation.

\noindent 2\textsuperscript{nd} order ODE $\therefore$ introduce 2 new variables.

\begin{align*}
    x_1 &= x\\
    x_2 &= \dot{x}\\
    \dot{x_2} &= \ddot{x}\\
    \underbar{$x$} &= 
    \begin{bmatrix}
        x_1 \cr
        x_2 \cr
    \end{bmatrix}\\
    \underbar{$\dot{x}$} &= 
    \begin{bmatrix}
        \dot{x_1} \cr
        \dot{x_2} \cr
    \end{bmatrix}\\
    \underbar{$\dot{x}$} &=
    \begin{bmatrix}
        x_2 \cr
        \dot{x_2} \cr
    \end{bmatrix}
\end{align*}
\begin{align}
    \underbar{$\dot{x}$} &=
    \begin{bmatrix}
        x_2 \cr
        \left( \frac{5}{3m} \right)
        \left( mg\sin \left( \phi \right)
        + c \left( \frac{{ V \left( 1 - e^{\frac{-R}{L}t}\right) }}{{ R \left( \delta - x \right) }} \right)^2
        - k \left( x - d \right)
        - b \dot{x} \right)\cr
    \end{bmatrix}
\end{align}

\bigskip
\noindent The states of the system are
$
    \underbar{$x$} = 
    \begin{bmatrix}
    x_1 \cr
    x_2 \cr
    \end{bmatrix}
$.\\
The input to the system is $V$.
