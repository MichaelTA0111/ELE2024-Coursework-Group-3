\subsection{Problem C2}\label{sec:qc2}\hfill

\medskip
\noindent Determine the Laplace transform of $f(t) = |cos(\omega t)|, t \geq 0, \omega > 0$.

\begin{align}
    e^{j \omega t} &= cos(\omega t) + j sin(\omega t) \cr
    \lap \{ e^{j \omega t} \} &= \lap \{cos(\omega t) + j sin(\omega t) \} \cr
    \frac{1}{s - j \omega} &= \lap \{ cos \omega \} + j \lap \{ sin \omega \} \cr
    \frac{s + j }{s^2 + \omega^2} &= \lap \{ cos \omega \} + j \lap \{ sin \omega \} \cr
    \frac{s}{s^2 + \omega^2} + j \frac{\omega}{s^2 + \omega^2} &= \lap \{ cos \omega \} + j \lap \{ sin \omega \} \cr
    \lap \{ cos \omega \} &= \frac{s}{s^2 + \omega^2} \\
    \lap \{ sin \omega \} &= \frac{\omega}{s^2 + \omega^2}
\end{align}

\medskip
\noindent The laplace transform of $f(t) = |cos(\omega t)|, t \geq 0, \omega > 0$ can be determined using equation (47):
\begin{align}
    \lap \{ f(t) \} &= \lap \{ |cos(\omega t)| \} \cr
    \lap \{ f(t) \} &= \frac{s}{s^2 + \omega^2}
\end{align}
