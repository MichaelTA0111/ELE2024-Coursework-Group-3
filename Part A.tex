\section{Problem A1}

\begin{equation}m\ddot{x} = mg\sin \left( \phi \right)  
+ F_{magnet} 
- k \left( x - d \right)  
- b \left( \dot{x} \right)  
+ F_{friction}
\end{equation}

\begin{equation}F_{mag} = c \frac{I^2}{y^2}\end{equation}

We need to find equations for $I$ and $y$.
\begin{equation}y = \delta - x\end{equation}
Equation for $y$ found.

\begin{equation}V = IR + \dot{I}L \Longrightarrow V = IR + \frac{dI}{dt}L\end{equation}

\begin{equation}V - IR = \frac{dI}{dt}L \Longrightarrow 
\frac{V - IR}{dI} = \frac{L}{dt} \Longrightarrow 
\frac{1}{V - IR}dI = \frac{1}{L}dt\end{equation}

\begin{equation}\int \frac{1}{V - IR}dI = \int \frac{1}{L}dt\end{equation}

Let $u = V - IR$,
\begin{equation}\frac{du}{dI} = -R\end{equation}
\begin{equation}dI = \frac{1}{-R} du\end{equation}

\begin{equation}\int \frac{1}{u} \times \frac{1}{-R} du= \int \frac{1}{L} dt\end{equation}
\begin{equation}\frac{1}{-R} \ln \left( u \right)  = \frac{1}{L}t + \sigma\end{equation}
where $\sigma$ is the constant of integration
\begin{equation}\frac{1}{-R} \ln \left( V - IR \right)  = \frac{1}{L}t + \sigma\end{equation}
\begin{equation} [\ln \left( V - IR \right)  = \frac{-R}{L}t + \sigma\end{equation}
\begin{equation} [\ln \left( V - IR \right) ]^{i}_{0} = [\frac{-R}{L}t + \sigma]^{T}_{0}\end{equation}
\begin{equation}\ln \left( V - iR \right)  - \ln \left( V \right)  = \frac{-R}{L}T\end{equation}
\begin{equation}\ln \left( V - IR \right)  - \ln \left( V \right)  = \frac{-R}{L}t\end{equation}
\begin{equation}\ln \left( \frac{V - IR}{V} \right)  = \frac{-R}{L}t\end{equation}
\begin{equation}\frac{V - IR}{V} = e^{\frac{-R}{L}t}\end{equation}

\[V - IR = Ve^{\frac{-R}{L}t}\]
\[V - Ve^{\frac{-R}{L}t} = IR\]
\[I = \frac{V}{R} - \frac{V}{R}e^{\frac{-R}{L}t}\]
Equation for $I$ found.

\bigskip
\[F_{mag} = c \frac{{ \left( \frac{V}{R} - \frac{V}{R}e^{\frac{-R}{L}t} \right) }^2}{{ \left( \delta - x \right) }^2}\]

Now we need to find an equation for $F_{friction}$
\bigskip
\[-F_{friction} \times r = - I_{inertia} \ddot{\theta}\]
"Pizza Slice theorem":
\[x = \theta r\]
Intertia of a Solid Sphere:
\[I_{inertia} = \frac{2}{5}mr^2\]
\[F_{friction} \times r =\frac{2}{5}mr^2 \times \frac{\ddot{x}}{r}\]
\[F_{friction} \times r =\frac{2}{5}mr  \ddot{x}\]
\[F_{friction} =\frac{2}{5}m  \ddot{x}\]
Equation for $F_{friction}$ found.
