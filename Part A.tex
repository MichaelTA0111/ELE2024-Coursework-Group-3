\subsection{Problem A1}

\begin{equation}
m\ddot{x} = 
mg\sin \left( \phi \right)  
+ F_{magnet} 
- k \left( x - d \right)  
- b \left( \dot{x} \right)  
+ F_{friction}
\end{equation}

\begin{equation}F_{mag} = c \frac{I^2}{y^2}\end{equation}

We need to find equations for $I$ and $y$.
\begin{equation}y = \delta - x\end{equation}
Equation for $y$ found.

\begin{equation*}V = IR + \dot{I}L \Longrightarrow V = IR + \frac{dI}{dt}L\end{equation*}

\begin{equation*}V - IR = \frac{dI}{dt}L \Longrightarrow 
\frac{V - IR}{dI} = \frac{L}{dt} \Longrightarrow 
\frac{1}{V - IR}dI = \frac{1}{L}dt\end{equation*}

\begin{equation*}\int \frac{1}{V - IR}dI = \int \frac{1}{L}dt\end{equation*}

Let $u = V - IR$,
\begin{equation*}\frac{du}{dI} = -R\end{equation*}
\begin{equation*}dI = \frac{1}{-R} du\end{equation*}

\begin{equation*}\int \frac{1}{u} \times \frac{1}{-R} du= \int \frac{1}{L} dt\end{equation*}
\begin{equation*}\frac{1}{-R} \ln \left( u \right)  = \frac{1}{L}t + \sigma\end{equation*}
where $\sigma$ is the constant of integration
\begin{equation*}\frac{1}{-R} \ln \left( V - IR \right)  = \frac{1}{L}t + \sigma\end{equation*}
\begin{equation*} [\ln \left( V - IR \right)  = \frac{-R}{L}t + \sigma\end{equation*}
\begin{equation*} [\ln \left( V - IR \right) ]^{i}_{0} = [\frac{-R}{L}t + \sigma]^{T}_{0}\end{equation*}
\begin{equation*}\ln \left( V - iR \right)  - \ln \left( V \right)  = \frac{-R}{L}T\end{equation*}
\begin{equation*}\ln \left( V - IR \right)  - \ln \left( V \right)  = \frac{-R}{L}t\end{equation*}
\begin{equation*}\ln \left( \frac{V - IR}{V} \right)  = \frac{-R}{L}t\end{equation*}
\begin{equation*}\frac{V - IR}{V} = e^{\frac{-R}{L}t}\end{equation*}

\begin{equation*}V - IR = Ve^{\frac{-R}{L}t}\end{equation*}
\begin{equation*}V - Ve^{\frac{-R}{L}t} = IR\end{equation*}
\begin{equation}I = \frac{V}{R} - \frac{V}{R}e^{\frac{-R}{L}t}\end{equation}
Equation for $I$ found.

\bigskip
\begin{equation*}F_{mag} = c \frac{{ \left( \frac{V}{R} - \frac{V}{R}e^{\frac{-R}{L}t} \right) }^2}{{ \left( \delta - x \right) }^2}\end{equation*}

Now we need to find an equation for $F_{friction}$
\bigskip
\begin{equation*}-F_{friction} \times r = - I_{inertia} \ddot{\theta}\end{equation*}
"Pizza Slice theorem":
\begin{equation*}x = \theta r\end{equation*}
Intertia of a Solid Sphere:
\begin{equation*}I_{inertia} = \frac{2}{5}mr^2\end{equation*}
\begin{equation*}F_{friction} \times r =\frac{2}{5}mr^2 \times \frac{\ddot{x}}{r}\end{equation*}
\begin{equation*}F_{friction} \times r =\frac{2}{5}mr  \ddot{x}\end{equation*}
\begin{equation}F_{friction} =\frac{2}{5}m  \ddot{x}\end{equation}
Equation for $F_{friction}$ found.

\subsection{Problem A4}
Linearise $f \left( \underbar{$\dot{\bar{x}}$}, \bar{V} \right) $ at $\underbar{$x$}^e$,$V^{e}$ as time $t\to \infty$

\bigskip
To Linearise $f$, $\psi = \frac{{ \left( \frac{V}{R} 
- \frac{V}{R}e^{\frac{-R}{L}t} \right) }^2}{{ \left( \delta - x_{1} \right) }^2} $ needs to be Linearised.

Taylors Theorm is used for Linearisation. It is modified a little to account for $t \to \infty$ .
\begin{equation}
\psi \left( x_{1},V \right)  \approx 
 \lim_{t\to\infty}
\left(
\psi \left( x_{1}^{e},V^{e} \right) 
+ \frac{\partial \psi}{\partial x_{1}}
%\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) }
\Bigr|_{\substack{ ( x_{1}^{e},V^{e}} ) }
 \left( x_{1} - x_{1}^e \right)  
+ \frac{\partial \psi}{\partial V}
%\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) } \left( V - V^{e} \right) \]
\Bigr|_{\substack{ ( x_{1}^{e},V^{e} ) }} ( V - V^{e} ) 
\right)
\end{equation}

\begin{equation}
\psi \left( x_{1},V \right)  \approx 
 \lim_{t\to\infty}
\psi \left( x_{1}^{e},V^{e} \right) 
+  \lim_{t\to\infty}
\frac{\partial \psi}{\partial x_{1}}
%\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) }
\Bigr|_{\substack{ ( x_{1}^{e},V^{e}} ) }
 \left( x_{1} - x_{1}^e \right)  
+  \lim_{t\to\infty}
\frac{\partial \psi}{\partial V}
%\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) } \left( V - V^{e} \right) \]
\Bigr|_{\substack{ ( x_{1}^{e},V^{e} ) }} ( V - V^{e} ) 
\end{equation}
 
 Find $\psi \left( x_{1},V \right)$.
 \begin{align*}
\lim_{t\to\infty}
 \psi \left( x_{1},V \right)  
 &=
\lim_{t\to\infty}
\frac{{ \left( \frac{V^{e}}{R} - \frac{V^{e}}{R}e^{\frac{-R}{L}t} \right) }^2}{{ \left( \delta - x_{1}^e \right) }^2}
\cr
&=
\frac{{ \lim_{t\to\infty}
\left( \frac{V^{e}}{R} - \frac{V^{e}}{R}e^{\frac{-R}{L}t} \right) }^2}{{ \left( \delta - x_{1}^e \right) }^2}
\cr
&=
\frac{{ \left( \frac{V^{e}}{R} - \frac{V^{e}}{R} \times 0 \right) }^2}{{ \left( \delta - x_{1}^e \right) }^2}
\cr
&=
\frac{{ \left( \frac{V^{e}}{R} \right) }^2}{{ \left( \delta - x_{1}^e \right) }^2}
\end{align*}

Find Each Derivative
\begin{align*}
\lim_{t\to\infty} \left(
\frac{\partial }{\partial x_{1}}
{ \left( \frac{V}{R} - \frac{V}{R}e^{\frac{-R}{L}t} \right) }^2

%\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) }
\Bigr|_{\substack{ ( x_{1}^{e},V^{e}} ) }

%\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) }

\right)
\cr

%\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) }
\Bigr|_{\substack{ ( x_{1}^{e},V^{e}} ) }
\right)
\cr
&=
\lim_{t\to\infty} \left(
\left( 2 \right)
\frac
{{ \left( \frac{V}{R} - \frac{V}{R}e^{\frac{-R}{L}t} \right) }^2}

%\Bigr|_{\substack{ \left( x_{1}^{e},V^{e}} \right) }
\Bigr|_{\substack{ ( x_{1}^{e},V^{e}} ) }
\right)
\cr
&=
\lim_{t\to\infty} \left(

\begin{align*}
